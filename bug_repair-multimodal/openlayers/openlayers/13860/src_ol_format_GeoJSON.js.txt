/**

function readMultiPolygonGeometry(object) {
  const coordinates = object['coordinates'];
  // Validate coordinates before creating the MultiPolygon
  if (!Array.isArray(coordinates)) {
    return new MultiPolygon([]);
  }
  
  // Ensure all coordinates are valid arrays
  const validCoordinates = coordinates.filter(function(polygonCoords) {
    return Array.isArray(polygonCoords) && polygonCoords.length > 0;
  });
  
  return new MultiPolygon(validCoordinates);
}
