from sympy.printing.printer import Printer
from sympy.tensor.indexed import Indexed

class MathMLContentPrinter(Printer):
    """Prints an expression to the Content MathML markup language.

    References: https://www.w3.org/TR/MathML2/chapter4.html
    """
    printmethod = '_mathml'

    def _print_Indexed(self, expr):
        """Print an Indexed object as MathML"""
        base = self._print(expr.base)
        indices = self._print(expr.indices[0])  # For single index
        apply = self.dom.createElement('apply')
        select = self.dom.createElement('select')
        apply.appendChild(select)
        apply.appendChild(base)
        apply.appendChild(indices)
        return apply
