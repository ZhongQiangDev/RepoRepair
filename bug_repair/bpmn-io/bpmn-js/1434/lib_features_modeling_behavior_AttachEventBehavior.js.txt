this.postExecuted('elements.create', LOW_PRIORITY, function(context) {
    var elements = context.elements;

    elements = elements.filter(function(shape) {
      var host = shape.host;

      return shouldReplace(shape, host);
    });

    if (elements.length !== 1) {
      return;
    }

    elements.forEach(function(element, index) {
      var host = element.host;

      if (host && isSubProcess(host)) {
        // Prevent pasting typed start events onto subprocess
        return;
      }

      context.elements[index] = self.replaceShape(element, host);
    });
  }, true);

function isSubProcess(element) {
  return isAny(element, ['bpmn:SubProcess']);
