  uploadTile_() {
    const helper = this.helper_;
    const gl = helper.getGL();
    const tile = this.tile;

    if (tile instanceof ImageTile || tile instanceof ReprojTile) {
      const texture = gl.createTexture();
      this.textures.push(texture);
      this.bandCount = 4;

      // Clip the gutter before uploading the texture
      const image = tile.getImage();
      const gutter = tile.getGutter();
      const pixelRatio = helper.getPixelRatio();
      const adjustedGutter = gutter * pixelRatio;

      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      canvas.width = image.width - 2 * adjustedGutter;
      canvas.height = image.height - 2 * adjustedGutter;
      ctx.drawImage(image, adjustedGutter, adjustedGutter, canvas.width, canvas.height, 0, 0, canvas.width, canvas.height);

      uploadImageTexture(gl, texture, canvas, tile.interpolate);
      return;
    }
