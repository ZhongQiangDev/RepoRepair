    },

source.on('load', function() {
  source.getSources().forEach(function(src) {
    src.getImage().then(function(img) {
      const metadata = img.getGDALMetadata(0);
      if (metadata.STATISTICS_MAXIMUM && metadata.STATISTICS_MINIMUM) {
        src.setMax(parseFloat(metadata.STATISTICS_MAXIMUM));
        src.setMin(parseFloat(metadata.STATISTICS_MINIMUM));
