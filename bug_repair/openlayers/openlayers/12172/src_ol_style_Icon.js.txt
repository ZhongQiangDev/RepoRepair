  /**
   * @return {Array<number>} Origin.
   * @api
   */
  getOrigin() {
    if (this.origin_) {
      return this.origin_;
    }
    let offset = this.offset_;
    const displacement = this.getDisplacement();
    const size = this.getSize();
    const iconImageSize = this.iconImage_.getSize();

    if (this.offsetOrigin_ != IconOrigin.TOP_LEFT) {
      if (!size || !iconImageSize) {
        return null;
      }
      offset = offset.slice();
      if (
        this.offsetOrigin_ == IconOrigin.TOP_RIGHT ||
        this.offsetOrigin_ == IconOrigin.BOTTOM_RIGHT
      ) {
        offset[0] = iconImageSize[0] - size[0] - offset[0];
      }
      if (
        this.offsetOrigin_ == IconOrigin.BOTTOM_LEFT ||
        this.offsetOrigin_ == IconOrigin.BOTTOM_RIGHT
      ) {
        offset[1] = iconImageSize[1] - size[1] - offset[1];
      }
    }
    offset[0] += displacement[0];
    offset[1] += displacement[1];

    // Ensure the origin is within the bounds of the icon image size
    if (size) {
      offset[0] = Math.max(0, Math.min(offset[0], iconImageSize[0] - size[0]));
      offset[1] = Math.max(0, Math.min(offset[1], iconImageSize[1] - size[1]));
    }

    this.origin_ = offset;
